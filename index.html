-- Fetching active users with transaction counts SELECT u.user_id, u.username, COUNT(t.transaction_id) AS total_orders, SUM(t.amount) AS lifetime_value FROM users u JOIN transactions t ON u.user_id = t.user_id WHERE u.status = 'active' GROUP BY u.user_id, u.username ORDER BY lifetime_value DESC LIMIT 100; COMMIT;
